### Private helpers (no docstring needed)

```python
def _end_ne(i: int, elements: list[Element]) -> tuple[float, float]:
    """End N/E of elements[i] — use next element's start, or integrate last."""
    if i < len(elements) - 1:
        return elements[i + 1].n, elements[i + 1].e
    state = calculate_point_on_element(
        elements[-1], elements[-1].sta_end - elements[-1].sta_start
    )
    return state.n, state.e

def _curve_center(n: float, e: float, azimuth_rad: float, k: float) -> tuple[float, float]:
    """Center of circular arc: perpendicular from start in direction of radius."""
    R = abs(1.0 / k)
    center_dir = azimuth_rad + math.pi / 2 if k > 0 else azimuth_rad - math.pi / 2
    return n + R * math.cos(center_dir), e + R * math.sin(center_dir)

def _rotation(k: float) -> str:   return "cw" if k > 0 else "ccw"

def _spiral_lx_type(transition: str) -> str:
    return {"CLOTHOID": "clothoid", "BLOSS": "bloss",
            "COSINE": "cosineCurve", "SINE": "sineCurve"}.get(transition, "clothoid")

def _coord(n: float, e: float) -> str:   return f"{n:.6f} {e:.6f}"
```

### Element → LandXML tag mapping

| Element.type | LandXML tag | Key attributes |
|---|---|---|
| T | `<Line>` | `length` |
| C | `<Curve>` | `rot`, `radius`, `length` + `<Start>` `<Center>` `<End>` |
| SPIN | `<Spiral>` | `radiusStart="INF"`, `radiusEnd=\|1/k_out\|`, `rot`, `spiType="toCurve"`, `type`, `length` |
| SPOUT | `<Spiral>` | `radiusStart=\|1/k_in\|`, `radiusEnd="INF"`, `rot`, `spiType="fromCurve"`, `type`, `length` |

Angle point (T ที่มาจาก R=0 PI): เป็น T element อยู่แล้ว → ออกเป็น `<Line>` ตามปกติ ไม่ต้องแยกกรณี

### Center formula (วิธีคิด)

ศูนย์กลางวงกลมอยู่ในทิศ perpendicular ของ entry tangent:
- k > 0 (เลี้ยวขวา, CW): `center_dir = azimuth_rad + π/2`
- k < 0 (เลี้ยวซ้าย, CCW): `center_dir = azimuth_rad - π/2`
- `center_N = elem.n + R · cos(center_dir)`  
- `center_E = elem.e + R · sin(center_dir)`

(WCB convention: dN = cos(az), dE = sin(az))

### XML skeleton

```xml
<?xml version="1.0" encoding="utf-8"?>
<LandXML version="1.2" date="YYYY-MM-DD" time="HH:MM:SS"
         xmlns="http://www.landxml.org/schema/LandXML-1.2">
  <Units>
    <Metric linearUnit="meter" areaUnit="squareMeter"
            volumeUnit="cubicMeter" angularUnit="decimal dd"
            directionUnit="decimal dd"/>
  </Units>
  <Alignments>
    <Alignment name="alignment" length="XXXX.XXXXXX" staStart="0.000000">
      <CoordGeom>
        <!-- one tag per Element -->
      </CoordGeom>
    </Alignment>
  </Alignments>
</LandXML>
```

ใช้ `ET.register_namespace('', 'http://www.landxml.org/schema/LandXML-1.2')` ก่อน build tree
เพื่อป้องกัน `ns0:` prefix ที่ Civil 3D ไม่ยอมรับ

ใช้ `ET.indent(root, space='  ')` (Python 3.9+) เพื่อ pretty-print

def _run_export_landxml(args: argparse.Namespace) -> int:
    vertices = _read_pi_table(args.alignment)
    if not vertices:
        raise ValueError('ไม่พบข้อมูล PI')
    result = build_alignment_from_pi(vertices)
    for issue in result.issues:

## 3. tests/test_landxml.py

Pattern: ใช้ `build_alignment_from_pi` กับ inline PI data (ไม่พึ่งไฟล์ disk)
ตรวจ XML string ด้วย `xml.etree.ElementTree.fromstring` + assertions

```python
import math, xml.etree.ElementTree as ET
import pytest
from smt.landxml import export_alignment_landxml
from smt.builders.alignment_builder import build_alignment_from_pi

NS = 'http://www.landxml.org/schema/LandXML-1.2'

def test_tangent_only():
    # build a 2-PI alignment (straight line)
    xml = export_alignment_landxml(...)
    root = ET.fromstring(xml)
    assert len(root.findall(f'.//{{{NS}}}Line')) == 1
    assert root.find(f'.//{{{NS}}}Curve') is None

def test_simple_curve():
    # 3-PI alignment with right-hand circular curve R=300
    xml = export_alignment_landxml(...)
    root = ET.fromstring(xml)
    curve = root.find(f'.//{{{NS}}}Curve')
    assert curve is not None
    center = curve.find(f'{{{NS}}}Center')
    cn, ce = [float(x) for x in center.text.split()]
    start = curve.find(f'{{{NS}}}Start')
    sn, se = [float(x) for x in start.text.split()]
    # center must be exactly R=300 from start
    assert math.isclose(math.hypot(cn - sn, ce - se), 300.0, abs_tol=1e-3)

def test_spiral_in():
    # alignment with Ls=60, R=400 CLOTHOID SPIN
    xml = export_alignment_landxml(...)
    root = ET.fromstring(xml)
    spiral = root.find(f'.//{{{NS}}}Spiral')
    assert spiral.get('radiusStart') == 'INF'
    assert math.isclose(float(spiral.get('radiusEnd')), 400.0, abs_tol=1e-6)
    assert spiral.get('spiType') == 'toCurve'

def test_angle_point():
    # R=0 PI → T element → only Line tags, no crash
    xml = export_alignment_landxml(...)
    root = ET.fromstring(xml)
    assert root.find(f'.//{{{NS}}}Curve') is None
    assert root.find(f'.//{{{NS}}}Spiral') is None
    assert len(root.findall(f'.//{{{NS}}}Line')) >= 1
```

## Risks & Notes

- `xml.etree.ElementTree` stdlib — ไม่เพิ่ม dependency
- `ET.indent()` ใช้ได้ Python 3.9+ (project ใช้ 3.10+)
- ถ้า Civil 3D ไม่รับ `"INF"` string → fallback เป็น `"9999999.000000"` (convention ทั่วไป) — ต้องทดสอบใน Civil 3D จริง
- `ET.register_namespace('', NS)` เรียกก่อน build tree เพื่อป้องกัน `ns0:` prefix
- Angle point (T element) ออกเป็น `<Line>` ตามปกติ ไม่ต้องแยกกรณีพิเศษ
- Alignment มี issues (overlap spiral ฯลฯ) → ยัง export ได้ แค่ print warning ไป stderr
